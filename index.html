let messages = []; let startTime = 0; let isPlaying = false; let interval; let currentSeconds = 0; let lastRenderedIndex = -1; const jsonInput = document.getElementById('jsonInput'); const playBtn = document.getElementById('playBtn'); const slider = document.getElementById('timer-slider'); const displayTime = document.getElementById('display-time'); const chatFlow = document.getElementById('chat-flow'); jsonInput.addEventListener('change', function(e) { const reader = new FileReader(); reader.onload = function(event) { try { const rawContent = event.target.result.trim(); const lines = rawContent.split('\n').filter(line => line.trim() !== ""); const validJsonArray = "[" + lines.join(",") + "]"; const rawData = JSON.parse(validJsonArray); messages = rawData.map(m => { const [h, min, s] = m.hora.split(':'); const d = new Date(); d.setHours(h, min, s, 0); return { date: d.getTime(), text: m.msg, userName: m.nick_name || "Sistema" // Mapeia o nome real capturado }; }); messages.sort((a, b) => a.date - b.date); if(messages.length > 0) { startTime = messages[0].date; const endTime = messages[messages.length - 1].date; slider.max = Math.floor((endTime - startTime) / 1000); resetToTime(0); document.getElementById('status-msg').innerText = "VOD CARREGADO - PRONTO"; } } catch(err) { alert("Erro ao ler arquivo: verifique se o formato do JSON está correto."); } }; reader.readAsText(e.target.files[0]); }); function formatTime(s) { const h = Math.floor(s / 3600); const m = Math.floor((s % 3600) / 60); const sec = s % 60; return `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${sec.toString().padStart(2,'0')}`; } function timeToSeconds(timeStr) { const parts = timeStr.split(':'); if (parts.length !== 3) return null; const h = parseInt(parts[0]) || 0; const m = parseInt(parts[1]) || 0; const s = parseInt(parts[2]) || 0; return (h * 3600) + (m * 60) + s; } function resetToTime(secs) { currentSeconds = Math.min(Math.max(0, secs), slider.max); slider.value = currentSeconds; displayTime.value = formatTime(currentSeconds); updateChat(true); } function updateChat(forceRefresh = false) { if (forceRefresh) { chatFlow.innerHTML = ''; lastRenderedIndex = -1; } const virtualTime = startTime + (currentSeconds * 1000); const fragment = document.createDocumentFragment(); let addedNew = false; for (let i = lastRenderedIndex + 1; i < messages.length; i++) { const m = messages[i]; if (m.date <= virtualTime) { // SÓ EXIBE SE TIVER TEXTO (evita balões vazios do log) if (m.text && m.text.trim() !== "") { const card = document.createElement('div'); card.className = 'msg-card'; card.innerHTML = `
${m.userName}
${m.text}
`; fragment.appendChild(card); addedNew = true; } lastRenderedIndex = i; } else { break; } } if (addedNew) { const status = document.getElementById('status-msg'); if(status) status.remove(); chatFlow.appendChild(fragment); chatFlow.scrollTop = chatFlow.scrollHeight; } } // Controles de Play/Pause playBtn.addEventListener('click', () => { if (messages.length === 0) return alert("Carregue o arquivo primeiro."); isPlaying = !isPlaying; playBtn.innerText = isPlaying ? "PAUSE" : "PLAY"; if(isPlaying) { interval = setInterval(() => { currentSeconds++; if (currentSeconds > slider.max) { clearInterval(interval); isPlaying = false; playBtn.innerText = "FIM"; return; } slider.value = currentSeconds; displayTime.value = formatTime(currentSeconds); updateChat(); }, 1000); } else { clearInterval(interval); } }); slider.addEventListener('input', () => { resetToTime(parseInt(slider.value)); }); displayTime.addEventListener('change', () => { const secs = timeToSeconds(displayTime.value); if (secs !== null) resetToTime(secs); else displayTime.value = formatTime(currentSeconds); });
